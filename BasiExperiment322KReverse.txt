void K_Reverse( List L, int K ){
    PtrToNode old,new,temp;
    int count = 1;
    new = L -> Next;
    old=new ->Next;
    while(count < K){
        temp = old ->Next;
        old ->Next = new;
        new = old;
        old = temp;
        count++;
    }
    L->Next->Next=old;
    L->Next=new;
}
